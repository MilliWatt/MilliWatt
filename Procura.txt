package teste;

import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.File;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;
import java.util.ArrayList;



public class Procura {

	public static File criaArquivos(File caminho, String nome) throws IOException{ 
	    File arquivo = new File(caminho, nome); 
	    arquivo.createNewFile(); 
	    return arquivo; 
	} 

	public static void escreverNoArquivo(ArrayList<String> linhas_com_href) throws IOException{ 
	    
		File diretorio = new File("C:\\Users\\Alunos\\workspace\\Teste\\");
	    File arquivo = criaArquivos(diretorio, "links_com_href.txt"); 
	      
	    BufferedWriter bw = new BufferedWriter(new FileWriter(arquivo, true)); 
	     
	    for(String linha : linhas_com_href){ 
	        bw.write(linha); 
	        bw.newLine(); 
	    } 
	   
	    bw.close(); 
	      
	}
	
	public static void main(String[] args) throws IOException {
		
		ArrayList<String> linhas = new ArrayList<String>();
		ArrayList<String> linhas_com_href = new ArrayList<String>();
		
		BufferedReader br = new BufferedReader(new FileReader("C:\\Users\\Alunos\\workspace\\Teste\\mw.txt"));
		
		while(br.ready()){
		   String linha = br.readLine();
		   linhas.add(linha);
		}  
		
		for(String linha : linhas){
			if(linha.contains("<a href=")){
				linhas_com_href.add(linha);
			}
		}
		
		escreverNoArquivo(linhas_com_href);
		
		for(String link : linhas_com_href){
			System.out.println(link+"\n");
		}
		
		
		
		br.close(); 
	}

}
